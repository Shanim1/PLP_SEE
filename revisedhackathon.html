Python Calculator

A simple calculator in Python that adds, subtracts, multiplies, or divides two numbers:

def calculator():
    op = input("Enter operation (+, -, *, /): ")
    a = float(input("First number: "))
    b = float(input("Second number: "))
    if op == '+':
        return a + b
    elif op == '-':
        return a - b
    elif op == '*':
        return a * b
    elif op == '/':
        return a / b
    else:
        return "Invalid operation"

print("Result:", calculator())
